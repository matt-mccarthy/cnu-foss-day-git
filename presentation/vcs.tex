\frame{
	\frametitle{What is a Version Control System?}

	Version control systems
	\begin{itemize}
		\item record revisions to ``things''
		\begin{itemize}
			\item documents
			\item code
			\item pictures and other media
		\end{itemize}
		\item label each revision
	\end{itemize}
	These two properties enable us to
	\begin{itemize}
		\item compare revisions
		\item restore previous revisions
		\item and merge different revisions
	\end{itemize}
}

\frame{
	\frametitle{Benefits of Version Control}
	\begin{itemize}
		\item \textit{Track} changes
		\item Provide some \textit{synchronization} capability
		\item Provide ways to \textit{compare} and \textit{revert} revisions
		\item Let many people edit the same files and \textit{merge} their changes
	\end{itemize}
}
